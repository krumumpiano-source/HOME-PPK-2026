### 9.1 Password Hashing

> ‚ö†Ô∏è **‡∏´‡∏°‡∏≤‡∏¢‡πÄ‡∏´‡∏ï‡∏∏:** Google Apps Script ‡πÑ‡∏°‡πà‡∏°‡∏µ bcrypt ‡πÉ‡∏ô‡∏ï‡∏±‡∏ß ‡πÉ‡∏ä‡πâ Utilities.computeDigest ‡πÅ‡∏ó‡∏ô

```javascript
function hashPassword(password) {
  const hash = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    password
  );
  return hash.map(byte => ('0' + (byte & 0xFF).toString(16)).slice(-2)).join('');
}

function verifyPassword(password, hash) {
  return hashPassword(password) === hash;
}
```

### 9.2 Session Management

```javascript
// ‡∏™‡∏£‡πâ‡∏≤‡∏á Token
function generateToken(userId) {
  const token = Utilities.getUuid();
  const expiry = new Date(Date.now() + 24 * 60 * 60 * 1000); // 24 hours
  
  // Store in PropertiesService
  const userProperties = PropertiesService.getUserProperties();
  userProperties.setProperty(token, JSON.stringify({
    userId: userId,
    expiry: expiry.toISOString()
  }));
  
  return token;
}

// ‡∏ï‡∏£‡∏ß‡∏à‡∏™‡∏≠‡∏ö Token
function validateToken(token) {
  const userProperties = PropertiesService.getUserProperties();
  const data = userProperties.getProperty(token);
  
  if (!data) return null;
  
  const parsed = JSON.parse(data);
  if (new Date() > new Date(parsed.expiry)) {
    userProperties.deleteProperty(token);
    return null;
  }
  
  return parsed.userId;
}
```
